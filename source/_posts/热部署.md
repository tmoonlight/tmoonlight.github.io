  1. 添加spring-boot-maven-plugin：  
  2. <build>  
  3.         <plugins>  
  4.             <plugin>  
  5.                 <groupId>org.springframework.boot</groupId>  
  6.                 <artifactId>spring-boot-maven-plugin</artifactId>  
  7.                 <configuration>  
  8.                     <!--fork :  如果没有该项配置，肯定devtools不会起作用，即应用不会restart \-->  
  9.                     <fork>true</fork>  
  10.                 </configuration>  
  11.             </plugin>  
  12.         </plugins>  
  13.    </build>
